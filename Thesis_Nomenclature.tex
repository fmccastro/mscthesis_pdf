%
% The definitions can be placed anywhere in the document body
% and their order is sorted by <symbol> automatically when
% calling makeindex in the makefile
%
% The \glossary command has the following syntax:
%
% \glossary{entry}
%
% The \nomenclature command has the following syntax:
%
% \nomenclature[<prefix>]{<symbol>}{<description>}
%
% where <prefix> is used for fine tuning the sort order,
% <symbol> is the symbol to be described, and <description> is
% the actual description.

% ----------------------------------------------------------------------
% Roman symbols [r]
\nomenclature[R]{$\mathbf{p}$}{Mobile robot position.}
\nomenclature[R]{$\mathbf{v}$}{Mobile robot linear velocity.}
\nomenclature[R]{$\mathcal{C},\, \mathcal{B},\, \mathcal{W}$}{Contact, body and world frames, respectively.}
\nomenclature[R]{$\mathbf{R}$}{Rotation matrix.}
\nomenclature[R]{x, y, z}{Position components.}
\nomenclature[R]{u, v, w}{Body frame linear velocity components.}
\nomenclature[R]{p, q, r}{Body frame angular velocity components.}
\nomenclature[R]{$n_i$}{Number of divisions of map on \textit{i} direction.}
\nomenclature[R]{$g(\cdot)$}{Equality constraints function.}
\nomenclature[R]{$g(\cdot)$}{Inequality constraints function.}
